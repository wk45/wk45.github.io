# make a function generating taus
def gen_taus():
    # make a list with starting point [0]
    taus = [0]
    
    # Since we can't determine the number of events in the time interval [0,100]
    # while loop is used here.
    while np.sum(taus) <= 100:
        tau = np.random.exponential(scale=10, size=1)
        taus.append(tau)
        
    return taus

fig = plt.figure()
num_events = list()

for i in range(0,5):
    
    taus = gen_taus()
    
    # Need to drop the last tau (inter-event time) 
    # since the cumulative sum of taus exceeds 100.
    taus = taus[:-1]
    
    # Generate t's (event times) by using taus
    t = np.cumsum(taus)
    num_events.append(len(taus)-1)
    
    # Draw a stair plot
    plt.step(t,range(0,len(taus)), 'o', where='post', alpha=0.5)
    
plt.xlim(0,100)    
plt.ylim(0,max(num_events)+0.5)
plt.xlabel('time')
plt.ylabel('# events')
plt.show()

# Collect Samples of {-1, +1} for 4 different T's.
samples = list()
for i in n:
    i = np.int(i)
    # draw samples from {0, 1}, x's.
    # Then we get {-1, +1} by (x - 0.5) * 2.
    draws = (np.random.binomial(1,0.5,size = i) - 1/2)*2
    samples.append(draws)
